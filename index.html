const noLaborables = [ "2025-12-24", "2025-12-31", "2026-12-24", "2026-12-31" ]; // Función para normalizar la fecha y evitar errores de zona horaria function parseFechaLocal(inputStr) { if (!inputStr) return null; let [y, m, d] = inputStr.split("-"); return new Date(y, m - 1, d); } function formatearFecha(fecha) { return fecha.toLocaleDateString('es-AR', { day: '2-digit', month: '2-digit', year: 'numeric' }); } function calcular() { let inputFecha = document.getElementById("fechaInicio").value; if (!inputFecha) return alert("Por favor, seleccione una fecha inicial."); let inicio = parseFechaLocal(inputFecha); let dias = parseInt(document.getElementById("dias").value); let tipo = document.querySelector('input[name="tipo"]:checked').value; if (isNaN(dias)) return alert("Por favor, ingrese la cantidad de días."); let fechaActual = new Date(inicio); let cont = 0; let sab = 0, dom = 0, fer = 0, nolab = 0; let aud = []; // Según la Ley de Procedimiento Administrativo, el plazo corre desde el día siguiente while (cont < dias) { fechaActual.setDate(fechaActual.getDate() + 1); let diaSemana = fechaActual.getDay(); // 0=Dom, 6=Sab let stringFecha = fechaActual.toISOString().split('T')[0]; let fVisual = formatearFecha(fechaActual); if (tipo === "corridos") { cont++; aud.push(`${fVisual} – contado (corrido)`); } else { // Lógica de Días Hábiles if (diaSemana === 0) { dom++; aud.push(`${fVisual} – domingo (excluido)`); } else if (diaSemana === 6) { sab++; aud.push(`${fVisual} – sábado (excluido)`); } else if (feriados.includes(stringFecha)) { fer++; aud.push(`${fVisual} – feriado (excluido)`); } else if (noLaborables.includes(stringFecha)) { nolab++; aud.push(`${fVisual} – asueto/no laborable (excluido)`); } else { cont++; aud.push(`${fVisual} – contado`); } } } // Mostrar resultados const resultadoTexto = formatearFecha(fechaActual); document.getElementById("resultado").innerText = resultadoTexto; document.getElementById("condicion").innerText = `${dias} días ${tipo}`; let detalleHTML = `Sábados: ${sab} | Domingos: ${dom}
Feriados/Asuetos: ${fer + nolab}`; document.getElementById("detalle").innerHTML = detalleHTML; document.getElementById("auditoria").innerHTML = aud.join("
"); // Actualizar área de impresión document.getElementById("printDatos").innerHTML = `Fecha inicial: ${formatearFecha(inicio)}
Plazo: ${dias} días ${tipo}`; document.getElementById("printResultado").innerText = resultadoTexto; document.getElementById("printResumen").innerHTML = detalleHTML; document.getElementById("printAuditoria").innerHTML = aud.join("
"); } function toggleAuditoria() { let a = document.getElementById("auditoria"); a.style.display = (a.style.display === "block") ? "none" : "block"; } function imprimir() { window.print(); }
